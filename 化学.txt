const QUIZ_SIZE = 10;
let quizList = [];
let current = 0;
let score = 0;
let streak = 0;
let maxStreak = 0;
let answered = false;
let typeScore = { 圧力: [0,0], 温度: [0,0], 物質量: [0,0] };

function shuffle(arr) {
  const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

function init() {
  // 3種類からバランスよく10問選ぶ（3-3-4）
  const byType = { 圧力: [], 温度: [], 物質量: [] };
  questions.forEach(q => byType[q.type].push(q));
  const s圧力 = shuffle(byType['圧力']).slice(0, 3);
  const s温度 = shuffle(byType['温度']).slice(0, 3);
  const s物質量 = shuffle(byType['物質量']).slice(0, 4);
  quizList = shuffle([...s圧力, ...s温度, ...s物質量]);

  current = 0; score = 0; streak = 0; maxStreak = 0; answered = false;
  typeScore = { 圧力: [0,0], 温度: [0,0], 物質量: [0,0] };
  document.getElementById('finish-card').style.display = 'none';
  document.getElementById('main-content').style.display = 'block';
  loadQuestion();
}

function loadQuestion() {
  const q = quizList[current];
  answered = false;
  document.getElementById('q-number').textContent = `Q${current + 1}`;
  document.getElementById('reaction').textContent = q.reaction;
  document.getElementById('condition').textContent = q.condition;
  document.getElementById('result-box').style.display = 'none';
  document.getElementById('progress-fill').style.width = `${(current / QUIZ_SIZE) * 100}%`;
  document.getElementById('progress-badge').textContent = `${current + 1}/${QUIZ_SIZE}`;
  document.getElementById('score-badge').textContent = `✅ ${score}`;

  const tl = document.getElementById('type-label');
  tl.textContent = q.type;
  tl.className = `type-badge type-${q.type}`;

  document.querySelectorAll('.choice-btn').forEach(b => {
    b.className = 'choice-btn';
    b.disabled = false;
  });
}

function select(choice) {
  if (answered) return;
  answered = true;
  const q = quizList[current];
  const correct = choice === q.answer;

  typeScore[q.type][1]++;
  if (correct) { score++; streak++; maxStreak = Math.max(maxStreak, streak); typeScore[q.type][0]++; }
  else { streak = 0; }

  document.querySelectorAll('.choice-btn').forEach(b => {
    b.disabled = true;
    if (b.textContent === q.answer) b.classList.add('correct');
    else if (b.textContent === choice) b.classList.add('wrong');
    else b.classList.add('dimmed');
  });

  const sb = document.getElementById('streak-badge');
  if (streak >= 2) { sb.style.display = 'inline'; sb.textContent = `🔥 ${streak}連続`; }
  else { sb.style.display = 'none'; }

  const rb = document.getElementById('result-box');
  rb.style.display = 'block';
  rb.style.background = correct ? '#064e3b' : '#450a0a';
  document.getElementById('result-emoji').textContent = correct ? '⭕' : '❌';
  document.getElementById('result-text').textContent = correct ? '正解！' : `不正解　正解: ${q.answer}`;
  document.getElementById('hint').textContent = q.hint;
  document.getElementById('next-btn').textContent = current + 1 >= QUIZ_SIZE ? '結果を見る' : '次の問題 →';
  document.getElementById('score-badge').textContent = `✅ ${score}`;
}

function next() {
  current++;
  if (current >= QUIZ_SIZE) showFinish();
  else loadQuestion();
}

function showFinish() {
  document.getElementById('main-content').style.display = 'none';
  const fc = document.getElementById('finish-card');
  fc.style.display = 'block';
  const pct = Math.round((score / QUIZ_SIZE) * 100);
  document.getElementById('finish-emoji').textContent = pct >= 80 ? '🏆' : pct >= 50 ? '📈' : '💪';
  document.getElementById('finish-score').textContent = `${score} / ${QUIZ_SIZE}`;
  document.getElementById('finish-pct').textContent = `${pct}%`;
  document.getElementById('finish-comment').textContent = pct >= 90 ? '完璧じゃん' : pct >= 70 ? 'いい感じ' : pct >= 50 ? 'もう少し' : '要復習';

  let bd = '';
  for (const [type, [ok, total]] of Object.entries(typeScore)) {
    if (total > 0) bd += `${type}: ${ok}/${total}　`;
  }
  document.getElementById('type-breakdown').textContent = bd;
  document.getElementById('streak-info').textContent = `最大連続正解: ${maxStreak}問`;
}

function restart() { init(); }
init();
</script>
